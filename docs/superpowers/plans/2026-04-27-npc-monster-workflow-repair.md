private promoteMiddleFilesToInput(
  files: PlainTextIngestionResult['files'],
  middleDir: string,
  inputDir: string,
): void {
  for (const file of files) {
    const sourcePath = join(middleDir, file.fileName);
    const targetPath = join(inputDir, file.fileName);
    mkdirSync(dirname(targetPath), { recursive: true });
    copyFileSync(sourcePath, targetPath);
  }
}

- [ ] **Step 5: Implement the minimal legacy-vs-structured guard**

In `YamlParser.applyField()`, before assigning `result.structuredActions`, add a helper check:

```ts
private isStructuredActionSection(value: unknown): boolean {
  if (!Array.isArray(value)) {
    return false;
  }

  return value.some((entry) => {
    if (!entry || typeof entry !== 'object' || Array.isArray(entry)) {
      return false;
    }

    return (
      '名称' in entry ||
      'name' in entry ||
      '类型' in entry ||
      'type' in entry ||
      '描述' in entry ||
      'describe' in entry ||
      '子活动' in entry ||
      '内嵌效果' in entry
    );
  });
}
```

Then wrap the structured parse block:

```ts
if (['特性', '动作', '附赠动作', '反应', '传奇动作'].includes(internalKey) && this.isStructuredActionSection(processedValue)) {
  const structuredParser = new StructuredActionParser();
  const sectionMap: Record<string, string> = {
    '特性': '特性',
    '动作': '动作',
    '附赠动作': '附赠动作',
    '反应': '反应',
    '传奇动作': '传奇动作',
  };
  const mapped = sectionMap[internalKey];
  if (mapped) {
    result.structuredActions = result.structuredActions ?? {};
    const sa = result.structuredActions as Record<string, StructuredActionData[]>;
    sa[mapped] = structuredParser.parseStructuredSection(processedValue, internalKey);
  }
}
```
